21  	public static final String BUILD_REPORT_JS = "/lib/buildReport.js";

24  	private final IOUtilsWrapper ioUtilsWrapper;

26  	public SpecRunnerExecutor(IOUtilsWrapper ioUtilsWrapper) {
27  		this.ioUtilsWrapper = ioUtilsWrapper;
28  	}
29  
30  	public SpecRunnerExecutor() {
31  		this(new IOUtilsWrapper());
32  	}

36  	public JasmineResult execute(URL runnerUrl, File junitXmlReport, WebDriver driver, int timeout, boolean debug, Log log, String format) {
37  		try {
38  			if (!(driver instanceof JavascriptExecutor)) {
39  				throw new RuntimeException("The provided web driver can't execute JavaScript: " + driver.getClass());
40  			}
41  			JavascriptExecutor executor = (JavascriptExecutor) driver;
42  			driver.get(runnerUrl.toString());
43  			this.waitForRunnerToFinish(driver, timeout, debug, log);
44  
45        		this.checkForConsoleErrors(driver, log);
46  
47  			JasmineResult jasmineResult = new JasmineResult();
48  			jasmineResult.setDetails(this.buildReport(executor,format));
49  			FileUtils.writeStringToFile(junitXmlReport, this.buildJunitXmlReport(executor,debug), "UTF-8");
50  
51  			return jasmineResult;
52  		} catch (Exception e) {
53  			throw new RuntimeException(e);
54  		} finally {
55  			try { 
56  				driver.quit(); 
57  			} catch (Exception e) {
58  				log.error("There was an exception quitting WebDriver.", e);
59  			}
60  		}
61  	}
62  
63    private void checkForConsoleErrors(WebDriver driver, Log log) {
64      WebElement head = driver.findElement(By.tagName("head"));
65      if (head != null) {
66        String jserrors = head.getAttribute("jmp_jserror");
67        if (StringUtils.isNotBlank(jserrors)) {
68          log.warn("JavaScript Console Errors:\n\n  * "+jserrors.replaceAll(":!:","\n  * ")+"\n\n");
69          throw new RuntimeException("There were javascript console errors.");
70        }
71      }
72    }
73  
74  	private String buildReport(JavascriptExecutor driver, String format) throws IOException {
75  		String script =
76  				this.ioUtilsWrapper.toString(BUILD_REPORT_JS) +
77  				"return jasmineMavenPlugin.printReport(window.jsApiReporter,{format:'"+format+"'});";
78  		Object report = driver.executeScript(script);
79  		return report.toString();
80  	}
81  
82  	private String buildJunitXmlReport(JavascriptExecutor driver, boolean debug) throws IOException {
83  		Object junitReport = driver.executeScript(
84  				this.ioUtilsWrapper.toString(CREATE_JUNIT_XML) +
85  				"return junitXmlReporter.report(window.jsApiReporter,"+debug+");");
86  		return junitReport.toString();
87  	}
88  
89  	private void waitForRunnerToFinish(final WebDriver driver, int timeout, boolean debug, Log log) throws InterruptedException {
90  		final JavascriptExecutor executor = (JavascriptExecutor) driver;
91  		new WebDriverWait(driver, timeout, 1000).until(new Predicate<WebDriver>() {
92  			@Override
93  			public boolean apply(WebDriver input) {
94  				return SpecRunnerExecutor.this.executionFinished(executor);
95  			}
96  		});
97  
98  		if (!this.executionFinished(executor)) {
99  			this.handleTimeout(timeout, debug, log);
100 		}
101 	}
102 
103 	private void handleTimeout(int timeout, boolean debug, Log log) {
104 		log.warn("Attempted to wait for your specs to finish processing over the course of " +
105 				timeout +
106 				" seconds, but it still appears to be running.");
107 		if(debug) {
108 			log.warn("Debug mode: will attempt to parse the incomplete spec runner results");
109 		} else {
110 			throw new IllegalStateException("Timeout occurred. Aborting execution of specs. (Try configuring 'debug' to 'true' for more details.)");
111 		}
112 	}
113 
114 	private Boolean executionFinished(JavascriptExecutor driver) {
115 		return (Boolean) driver.executeScript("return (window.jsApiReporter === undefined) ? false : window.jsApiReporter.finished");
116 	}
