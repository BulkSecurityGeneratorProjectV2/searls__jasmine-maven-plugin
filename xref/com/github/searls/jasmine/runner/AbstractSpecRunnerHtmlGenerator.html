3   import java.io.IOException;
4   import java.util.Collection;
5   import java.util.List;
6   import java.util.Set;

8   import org.codehaus.plexus.util.StringUtils;
9   import org.stringtemplate.v4.ST;
10  
11  import com.github.searls.jasmine.format.FormatsScriptTags;
12  
13  public abstract class AbstractSpecRunnerHtmlGenerator {
14    private static final String SOURCE_ENCODING = "sourceEncoding";
15    private static final String CSS_DEPENDENCIES_TEMPLATE_ATTR_NAME = "cssDependencies";
16    protected static final String JAVASCRIPT_DEPENDENCIES_TEMPLATE_ATTR_NAME = "javascriptDependencies";
17    protected static final String SOURCES_TEMPLATE_ATTR_NAME = "sources";
18    protected static final String REPORTER_ATTR_NAME = "reporter";
19    private final HtmlGeneratorConfiguration configuration;
20    private final FormatsScriptTags formatsScriptTags = new FormatsScriptTags();
21  
22    protected AbstractSpecRunnerHtmlGenerator(HtmlGeneratorConfiguration configuration) {
23      this.configuration = configuration;
24    }
25  
26    protected void setEncoding(HtmlGeneratorConfiguration htmlGeneratorConfiguration, ST template) {
27      template.add(SOURCE_ENCODING, StringUtils.isNotBlank(htmlGeneratorConfiguration.getSourceEncoding()) ? htmlGeneratorConfiguration.getSourceEncoding() : SpecRunnerHtmlGenerator.DEFAULT_SOURCE_ENCODING);
28    }
29  
30    protected ST resolveHtmlTemplate() throws IOException {
31      String htmlTemplate = configuration.getRunnerTemplate();
32      return new ST(htmlTemplate,'$','$');
33    }
34  
35    protected void applyCssToTemplate(List<String> styles, ST template) throws IOException {
36      StringBuilder css = new StringBuilder();
37      for (String cssFile : styles) {
38        css.append("<link rel=\"stylesheet\" type=\"text/css\" href=\"").append(cssFile).append("\"/>");
39      }
40      template.add(CSS_DEPENDENCIES_TEMPLATE_ATTR_NAME, css.toString());
41    }
42  
43    public HtmlGeneratorConfiguration getConfiguration() {
44      return configuration;
45    }
46  
47    protected void applyScriptTagsToTemplate(String sourcesTemplateAttrName, Collection<String> scripts, ST template) throws IOException {
48      template.add(sourcesTemplateAttrName, formatsScriptTags.format(scripts));
49    }
50  }
