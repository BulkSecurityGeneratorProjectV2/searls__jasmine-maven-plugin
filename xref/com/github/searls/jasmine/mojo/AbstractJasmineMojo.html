50  	 * before other sources (and specs) in a particular order. Each source will first be
51  	 * searched for relative to <code>${jsSrcDir}</code>, then <code>${jsTestSrcDir}</code>,

376 	@Override
377 	public void execute() throws MojoExecutionException, MojoFailureException {
378 
379 		this.loadResources();
380 
381 		this.sources = new ScriptSearch(this.jsSrcDir,this.sourceIncludes,this.sourceExcludes);
382 		this.specs = new ScriptSearch(this.jsTestSrcDir,this.specIncludes,this.specExcludes);
383 
384 		try {
385 			this.run();
386 		} catch(MojoFailureException e) {
387 			throw e;
388 		} catch(Exception e) {
389 			throw new MojoExecutionException("The jasmine-maven-plugin encountered an exception: \n"+this.stringifiesStackTraces.stringify(e),e);
390 		}
391 	}

501 	private File getResourceAsFile(String parameter, String resourceLocation) throws MojoExecutionException {
502 		File file = null;
503 
504 		if (resourceLocation != null) {
505 			this.locator.addSearchPath( "url", "" );
506 			this.locator.addSearchPath( FileResourceLoader.ID, this.mavenProject.getFile().getParentFile().getAbsolutePath() );
507 
508 			ClassLoader origLoader = Thread.currentThread().getContextClassLoader();
509 			try {
510 				Thread.currentThread().setContextClassLoader(this.getClass().getClassLoader());
511 				try {
512 					file = this.locator.getResourceAsFile(resourceLocation);
513 				} catch (Exception e) {
514 					throw new MojoExecutionException(String.format(ERROR_FILE_DNE,parameter,resourceLocation));
515 				}
516 			}
517 			finally {
518 				Thread.currentThread().setContextClassLoader( origLoader );
519 			}
520 		}
521 		return file;
522 	}
