13  import java.io.File;
14  import java.util.ArrayList;
15  import java.util.Arrays;
16  import java.util.List;
17  
18  import static org.hamcrest.Matchers.hasSize;
19  import static org.hamcrest.Matchers.is;
20  import static org.junit.Assert.assertThat;
21  import static org.mockito.Mockito.mock;
22  import static org.mockito.Mockito.when;
23  
24  @RunWith(MockitoJUnitRunner.class)
25  public class ReporterRetrieverTest {
26  
27    @Mock
28    ResourceRetriever resourceRetriever;
29  
30    @Mock
31    MavenProject mavenProject;
32  
33    File targetDir = new File(".");
34  
35    @Mock
36    File standardReporter;
37  
38    @Mock
39    File junitXmlReporter;
40  
41    ReporterRetriever subject;
42  
43    @Before
44    public void setUp() throws Exception {
45      subject = new ReporterRetriever(resourceRetriever);
46  
47      when(resourceRetriever.getResourceAsFile("reporter", ReporterRetriever.STANDARD_REPORTER, mavenProject)).thenReturn(standardReporter);
48      when(resourceRetriever.getResourceAsFile("reporter", ReporterRetriever.JUNIT_XML_REPORTER, mavenProject)).thenReturn(junitXmlReporter);
49    }
50  
51    @Test
52    public void itShouldRetrieveReporters() throws Exception {
53      String customReporterPath = "/foo/bar";
54      File customReporter = mock(File.class);
55      when(resourceRetriever.getResourceAsFile("reporter", customReporterPath, mavenProject)).thenReturn(customReporter);
56      List<Reporter> incomingReporters = Arrays.asList(new Reporter("STANDARD"), new Reporter(customReporterPath));
57  
58      List<Reporter> reporters = subject.retrieveReporters(incomingReporters, mavenProject);
59  
60      assertThat(reporters, hasSize(incomingReporters.size()));
61      assertThat(reporters.get(0).reporterFile, is(standardReporter));
62      assertThat(reporters.get(1).reporterFile, is(customReporter));
63    }
64  
65    @Test
66    public void itShouldRetrieveStandardReporterAsDefault() throws Exception {
67      List<Reporter> reporters = subject.retrieveReporters(new ArrayList<Reporter>(), mavenProject);
68  
69      assertThat(reporters, hasSize(1));
70      assertThat(reporters.get(0).reporterFile, is(standardReporter));
71    }
72  
73    @Test
74    public void itShouldRetrieveFileSystemReporters() throws Exception {
75      String customReporterPath = "/foo/bar";
76      File customReporter = mock(File.class);
77      when(resourceRetriever.getResourceAsFile("reporter", customReporterPath, mavenProject)).thenReturn(customReporter);
78      List<FileSystemReporter> incomingReporters = Arrays.asList(new FileSystemReporter("TEST-file.xml", "JUNIT_XML"), new FileSystemReporter("TEST-custom.file", customReporterPath));
79  
80      List<FileSystemReporter> reporters = subject.retrieveFileSystemReporters(incomingReporters, targetDir, mavenProject);
81  
82      assertThat(reporters, hasSize(incomingReporters.size()));
83      assertThat(reporters.get(0).reporterFile, is(junitXmlReporter));
84      assertThat(reporters.get(0).file.getAbsolutePath(), new EndsWith("TEST-file.xml"));
85      assertThat(reporters.get(1).reporterFile, is(customReporter));
86      assertThat(reporters.get(1).file.getAbsolutePath(), new EndsWith("TEST-custom.file"));
87    }
88  
89    @Test
90    public void itShouldRetrieveJUnitFileReporterAsDefault() throws Exception {
91      List<FileSystemReporter> reporters = subject.retrieveFileSystemReporters(new ArrayList<FileSystemReporter>(), targetDir, mavenProject);
92  
93      assertThat(reporters, hasSize(1));
94      assertThat(reporters.get(0).reporterFile, is(junitXmlReporter));
95      assertThat(reporters.get(0).file.getAbsolutePath(), new EndsWith("TEST-jasmine.xml"));
96    }
97  }
