22  @RunWith(PowerMockRunner.class)
23  public class WebJarResourceHandlerTest {
24    @Mock
25    private Request baseRequest;
26    @Mock
27    private HttpServletRequest request;
28    @Mock
29    private HttpServletResponse response;
30    @Mock
31    private Resource resource;
32    @Mock
33    private PrintWriter writer;
34  
35    private WebJarResourceHandler subject;
36  
37    @Before
38    public void before() {
39      String jquery = "src/test/resources/webjars/jquery-1.10.2.jar";
40      subject = new WebJarResourceHandler(projectClassLoaderOf(jquery));
41    }
42  
43    @Test
44    public void whenTargetContainsFileFromWebJarRespondWithResourceContent() throws Exception {
45      // given
46      when(baseRequest.isHandled()).thenReturn(false);
47      when(response.getWriter()).thenReturn(writer);
48  
49      // when
50      subject.handle("/jquery.js", baseRequest, request, response);
51  
52      // then
53      verify(writer).write(contains("jQuery JavaScript Library v1.10.2"));
54      verify(baseRequest).setHandled(true);
55    }
56  
57    @Test
58    public void whenResourceIsMissingThenDoNotProcess() throws Exception {
59      // given
60      when(baseRequest.isHandled()).thenReturn(false);
61      when(response.getWriter()).thenReturn(writer);
62  
63      // when
64      subject.handle("/notExistingResource", baseRequest, request, response);
65  
66      // then
67      verify(writer, never()).write(anyString());
68      verify(baseRequest, never()).setHandled(true);
69    }
