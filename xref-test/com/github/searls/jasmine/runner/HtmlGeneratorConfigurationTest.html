30  	@Test
31  	public void shouldReadCustomTemplateWhenOneIsProvided() throws IOException {
32  		mockStatic(FileUtils.class);
33  
34  		File expected = mock(File.class);
35  		IOUtilsWrapper ioUtilsWrapper = mock(IOUtilsWrapper.class);
36  		this.generatorConfiguration = this.initGenerator(ioUtilsWrapper,null,expected);

39  		verifyStatic(times(1));

43  	@Test
44  	public void shouldReadSpecRunnerTemplateWhenOneIsProvided() throws IOException {
45  		IOUtilsWrapper ioUtilsWrapper = mock(IOUtilsWrapper.class);
46  		this.generatorConfiguration = this.initGenerator(ioUtilsWrapper,SpecRunnerTemplate.REQUIRE_JS,null);
47  		this.generatorConfiguration.getRunnerTemplate();

71  	private HtmlGeneratorConfiguration initGenerator(IOUtilsWrapper ioUtilsWrapper, SpecRunnerTemplate template, File customTemplate) throws IOException {
72  		JasmineConfiguration mock = mock(JasmineConfiguration.class);
73  		when(mock.getSpecRunnerTemplate()).thenReturn(template);
74  		when(mock.getCustomRunnerTemplate()).thenReturn(customTemplate);
75  		return new HtmlGeneratorConfiguration(ioUtilsWrapper, ReporterType.JsApiReporter, mock, mock(ScriptResolver.class));
76  	}
